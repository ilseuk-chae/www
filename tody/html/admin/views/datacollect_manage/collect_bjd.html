            -- 
        TRUNCATE TABLE SIDO_EMD_CODE;

        LOAD DATA LOCAL INFILE 'C:/temp/SIDO_EMD_CODE.csv'
        INTO TABLE SIDO_EMD_CODE
        CHARACTER SET utf8mb4
        FIELDS TERMINATED BY ','
        ENCLOSED BY '"'
        LINES TERMINATED BY '\r\n'
        IGNORE 1 ROWS
        (
            @emd_code,
            @sgg_code,
            @sido_code,
            @emd_name,
            @sgg_name,
            @sido_name,
            @use_yn
        )
        SET
            emd_code = TRIM(@emd_code),
            sgg_code = TRIM(@sgg_code),
            sido_code = TRIM(@sido_code),
            emd_name = TRIM(@emd_name),
            sgg_name = TRIM(@sgg_name),
            sido_name = TRIM(@sido_name),

            -- 🔥 핵심: 1 → Y 변환
            use_yn = CASE 
                WHEN @use_yn = '1' THEN 'Y'
                WHEN @use_yn = '0' THEN 'N'
                ELSE 'Y'
            END,

            id = NULL;   -- 🔥 필수
