<실행 순서>

  1. DB 적재 — CSV 파일을 DB 테이블에 적재합니다.

<방법>

  1. 다운받은 CSV 파일(AL_D195_xx_yyyymmdd)을 land_characteristics_tmp 테이블에 저장합니다.
    -- 이전 테이블 삭제
        TRUNCATE TABLE land_characteristics_tmp;
        
        -- CSV 파일 적재
        LOAD DATA LOCAL INFILE 'E:/source/Tody/AL_D002/AL_D195_41_20260308.csv'  -- 영문 경로로 변경
        IGNORE
        INTO TABLE land_characteristics_tmp
        CHARACTER SET euckr
        FIELDS TERMINATED BY ',' 
        OPTIONALLY ENCLOSED BY '"'
        LINES TERMINATED BY '\n'
        IGNORE 1 ROWS
        (
                pnu,
                ldCode,
                ldCodeNm,
                regstrSeCode,
                regstrSeCodeNm,
                mnnmSlno,
                ladSn,
                stdrYear,
                stdrMt,
                lndcgrCode,
                lndcgrCodeNm,
                lndpclAr,
                prposArea1,
                prposArea1Nm,
                prposArea2,
                prposArea2Nm,
                ladUseSittn,
                ladUseSittnNm,
                tpgrphHgCode,
                tpgrphHgCodeNm,
                tpgrphFrmCode,
                tpgrphFrmCodeNm,
                roadSideCode,
                roadSideCodeNm,
                pblntfPclnd,
                lastUpdtDt
        );

        -- 데이터 개수 확인
        SELECT COUNT(*) FROM land_characteristics_tmp;
  2. land_characteristics_tmp 를 land_characteristics_xx 에 복사합니다. 
    
        SET autocommit = 0;
  
        INSERT INTO land_characteristics_41
        SELECT * FROM land_characteristics_tmp;
        
        COMMIT;
